1 Algroithm: Depth-First Search 
 2 Input: nodeList the list of the node, pathList the list of the path

 4 function Set DFS

 9         m <- n.getMark
10         LinkedList linkedNode <- empty    
11         for i = 0 to pathList.size
12             p <- value in pathList with index i
13             a <- p.getA
14             b <- p.getB
15             if m is equal to a
16                 add b to linkedNode
17             end if
18             else if m is equal to b
19                 add a to linkedNode
20                  end if
21         end for
22         if linkedNode is not empty
23             add an entry(m, linkedNode) into graph         
24         end if
25     end while
26     start <- 0
27     LinkedList output <- empty
28     stack s <- empty
29     push start to s 
30         while s is not empty
31             the value of pop s sign as top
32             add <- 1
33             for i = 1 to output.size
34                 if top is equal to the value in output list with index i
35                     add <- 0
36                     break
37                 end if
38             end for
39             if add is equal to 1
40                 add top to output list 
41             end if
42             for each value i in graph with index top       
43                 if i is not in the output list
44                     push i to s

47         end while
